add_library(omrvmstartup STATIC
	omrvmstartup.cpp
	omrrasinit.c
	omrtrcinit.c
)

# We need some tracegen files from omrcore
target_include_directories(omrvmstartup
	PRIVATE
		${omr_BINARY_DIR}/omr
)
add_dependencies(omrvmstartup
	trc_omrvm
	trc_omrti
)


target_link_libraries(omrvmstartup
	PRIVATE
		omrcore
		${OMR_HOOK_LIB}
		${OMR_TRACE_LIB}
)

if(OMR_GC)
	target_link_libraries(omrvmstartup
		PRIVATE
			${OMR_GC_LIB}
	)
endif()
